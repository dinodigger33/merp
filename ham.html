.search-bar { width: 300px; height: 40px; background-color: #fff; border: 1px solid #ccc; border-radius: 20px; position: absolute; top: 100px; left: 100px; display: flex; align-items: center; padding: 0 10px; box-sizing: border-box; cursor: grab; } #searchInput { width: 100%; border: none; outline: none; padding: 5px; border-radius: 15px; font-size: 14px; } const searchBar = document.getElementById('searchBar'); const searchInput = document.getElementById('searchInput'); let offsetX, offsetY; let isDragging = false; let velocityX = 2; // Horizontal speed let velocityY = 2; // Vertical speed const friction = 0.98; // Friction factor to slow down bouncing // Function to handle dragging function onMouseMove(e) { if (isDragging) { searchBar.style.left = `${e.clientX - offsetX}px`; searchBar.style.top = `${e.clientY - offsetY}px`; } } // Mouse down event to start dragging searchBar.addEventListener('mousedown', (e) => { isDragging = true; offsetX = e.clientX - searchBar.offsetLeft; offsetY = e.clientY - searchBar.offsetTop; document.addEventListener('mousemove', onMouseMove); }); // Mouse up event to stop dragging document.addEventListener('mouseup', () => { isDragging = false; document.removeEventListener('mousemove', onMouseMove); }); // Function to handle bouncing function bounce() { if (!isDragging) { let rect = searchBar.getBoundingClientRect(); // Update position searchBar.style.left = `${rect.left + velocityX}px`; searchBar.style.top = `${rect.top + velocityY}px`; // Check for collision with window edges if (rect.right >= window.innerWidth || rect.left <= 0) { velocityX = -velocityX * friction; } if (rect.bottom >= window.innerHeight || rect.top <= 0) { velocityY = -velocityY * friction; } // Apply friction velocityX *= friction; velocityY *= friction; } requestAnimationFrame(bounce); } // Start the bouncing animation bounce();
