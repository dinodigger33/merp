.tab-container { display: flex; cursor: move; } .tab { padding: 10px; background-color: #ccc; margin-right: 5px; user-select: none; } .content-container { border: 1px solid #ccc; padding: 10px; margin-top: 10px; resize: both; overflow: auto; } .content { display: none; } .content.active { display: block; } const tabs = document.querySelectorAll('.tab'); const contents = document.querySelectorAll('.content'); let currentTab = null; tabs.forEach((tab, index) => { tab.addEventListener('click', () => { // Remove active class from all contents contents.forEach(content => content.classList.remove('active')); // Add active class to the clicked tab's content contents[index].classList.add('active'); }); tab.addEventListener('dragstart', (e) => { currentTab = tab; e.dataTransfer.effectAllowed = 'move'; }); tab.addEventListener('dragover', (e) => { e.preventDefault(); }); tab.addEventListener('drop', (e) => { e.preventDefault(); if (currentTab !== tab) { const container = document.querySelector('.tab-container'); const tabsArray = Array.from(tabs); const currentIndex = tabsArray.indexOf(currentTab); const targetIndex = tabsArray.indexOf(tab); if (currentIndex < targetIndex) { container.insertBefore(currentTab, tab.nextSibling); } else { container.insertBefore(currentTab, tab); } } }); });
