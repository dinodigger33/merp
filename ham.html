.window { width: 200px; height: 150px; background-color: lightblue; position: absolute; top: 100px; left: 100px; cursor: grab; /* Additional styling as needed */ } const draggableWindow = document.getElementById('draggableWindow'); let offsetX, offsetY; let isDragging = false; let velocityX = 2; // Horizontal speed let velocityY = 2; // Vertical speed const friction = 0.98; // Friction factor to slow down bouncing // Function to handle dragging function onMouseMove(e) { if (isDragging) { draggableWindow.style.left = `${e.clientX - offsetX}px`; draggableWindow.style.top = `${e.clientY - offsetY}px`; } } // Mouse down event to start dragging draggableWindow.addEventListener('mousedown', (e) => { isDragging = true; offsetX = e.clientX - draggableWindow.offsetLeft; offsetY = e.clientY - draggableWindow.offsetTop; document.addEventListener('mousemove', onMouseMove); }); // Mouse up event to stop dragging document.addEventListener('mouseup', () => { isDragging = false; document.removeEventListener('mousemove', onMouseMove); }); // Function to handle bouncing function bounce() { if (!isDragging) { let rect = draggableWindow.getBoundingClientRect(); // Update position draggableWindow.style.left = `${rect.left + velocityX}px`; draggableWindow.style.top = `${rect.top + velocityY}px`; // Check for collision with window edges if (rect.right >= window.innerWidth || rect.left <= 0) { velocityX = -velocityX * friction; } if (rect.bottom >= window.innerHeight || rect.top <= 0) { velocityY = -velocityY * friction; } // Apply friction velocityX *= friction; velocityY *= friction; } requestAnimationFrame(bounce); } // Start the bouncing animation bounce();
